# .htaccess - Apache configuration file
# This file tells the web server how to handle different file types

# Enable CGI in this folder
Options +ExecCGI
# This allows us to run CGI programs (our .cgi files)

AddHandler cgi-script .cgi
# This tells Apache that .cgi files are CGI scripts to execute

# Security headers
Header set X-Content-Type-Options nosniff
# Prevents browsers from guessing content types (security)

Header set X-Frame-Options SAMEORIGIN
# Prevents site from being embedded in frames on other sites

Header set X-XSS-Protection "1; mode=block"
# Enables browser's XSS protection

# Set index.cgi as the default page
DirectoryIndex index.cgi
# When someone visits the directory, show index.cgi automatically

# Force .cpp files to display as plain text
<FilesMatch "\.cpp$">
  RemoveHandler .cpp
  # Don't try to execute .cpp files
  
  AddType text/plain .cpp
  # Display .cpp files as plain text in browser
  
  Require all granted
  # Allow everyone to view .cpp files
</FilesMatch>
